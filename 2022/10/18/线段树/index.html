线段树

1.单点修改，区间查询线段树

#include<bits/stdc++.h>
using namespace std;
const int N = 2e5 + 10;

struct info
{
	int minv, cnt_min;
};

struct Node
{
	info val;
}tr[N * 4];
int n, q;
int a[N];

info operator + (const info &l, const info &r)
{
	info a;
	a.minv = min(l.minv, r.minv);
	if(l.minv == r.minv)	a.cnt_min = l.cnt_min + r.cnt_min;
	else if(l.minv < r.minv)	a.cnt_min = l.cnt_min;
	else	 a.cnt_min = r.cnt_min;
	return a;
}

void pushup(int u)
{
	tr[u].val = tr[u << 1].val + tr[u << 1 | 1].val;
}

void build(int u, int l, int r)
{
	if(l == r)	tr[u].val = {a[l], 1};
	else
	{
		int mid = l + r >> 1;
		build(u << 1, l, mid);
		build(u << 1 | 1, mid + 1, r);
		pushup(u);
	}
}

void change(int u, int l, int r, int pos, int val)
{
	if(l == r)	tr[u].val = {val, 1};
	else
	{
		int mid = l + r >> 1;
		if(pos <= mid)	change(u << 1, l, mid, pos, val);
		else	change(u << 1 | 1, mid + 1, r, pos, val);
		pushup(u);
	}
}

info query(int u, int l, int r, int ql, int qr)
{
	if(l == ql && r == qr)	return tr[u].val;
	int mid = l + r >> 1;
	if(qr <= mid)	return query(u << 1, l, mid, ql, qr);
	else if(ql > mid)	return query(u << 1 | 1, mid + 1, r, ql, qr);
	else
	{
		return query(u << 1, l, mid, ql, mid) + query(u << 1 | 1, mid + 1, r, mid + 1, qr);
	}
	
}

int main()
{
	scanf("%d%d", &n, &q);
	for(int i = 1; i <= n; i ++)	scanf("%d", &a[i]);
	build(1, 1, n);
	while(q --)
	{
		int a, b, c;
		scanf("%d%d%d", &a, &b, &c);
		if(a == 1)
		{
			change(1, 1, n, b, c);
		}
		else
		{
			auto a = query(1, 1, n, b, c);
			printf("%d %d\n", a.minv, a.cnt_min);
		}
	}
}

2.维护复杂信息的线段树（最大子段和）

#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const int N = 2e5 + 10;

struct info
{
	ll mss, mpre, msuf, s;
	info () {}
	info (int a):mss(a), mpre(a), msuf(a), s(a) {}
};

struct Node
{
	info val;
}tr[N * 4];
int n, q;
int a[N];

info operator + (const info &l, const info &r)
{
	info a;
	a.mss = max(max(l.mss, r.mss), l.msuf + r.mpre);
	a.mpre = max(l.mpre, l.s + r.mpre);
	a.msuf = max(r.msuf, r.s + l.msuf);
	a.s = l.s + r.s;
	return a;
}

void pushup(int u)
{
	tr[u].val = tr[u << 1].val + tr[u << 1 | 1].val;
}

void build(int u, int l, int r)
{
	if(l == r)	tr[u].val = info(a[l]);
	else
	{
		int mid = l + r >> 1;
		build(u << 1, l, mid);
		build(u << 1 | 1, mid + 1, r);
		pushup(u);
	}
}

void change(int u, int l, int r, int pos, int val)
{
	if(l == r)	tr[u].val = info(val);
	else
	{
		int mid = l + r >> 1;
		if(pos <= mid)	change(u << 1, l, mid, pos, val);
		else	change(u << 1 | 1, mid + 1, r, pos, val);
		pushup(u);
	}
}

info query(int u, int l, int r, int ql, int qr)
{
	if(l == ql && r == qr)	return tr[u].val;
	int mid = l + r >> 1;
	if(qr <= mid)	return query(u << 1, l, mid, ql, qr);
	else if(ql > mid)	return query(u << 1 | 1, mid + 1, r, ql, qr);
	else
	{
		return query(u << 1, l, mid, ql, mid) + query(u << 1 | 1, mid + 1, r, mid + 1, qr);
	}
	
}

int main()
{
	scanf("%d%d", &n, &q);
	for(int i = 1; i <= n; i ++)	scanf("%d", &a[i]);
	build(1, 1, n);
	while(q --)
	{
		ll a, b, c;
		scanf("%lld%lld%lld", &a, &b, &c);
		if(a == 1)
		{
			change(1, 1, n, b, c);
		}
		else
		{
			auto a = query(1, 1, n, b, c);
			printf("%lld\n", a.mss);
		}
	}
}

3.区间修改，区间查询线段树

#include<bits/stdc++.h>
using namespace std;
const int N = 2e5 + 10;
typedef long long ll;
ll n, q;
ll a[N];

struct info
{
	ll maxv;
};

struct tag
{
	ll add;
};

info operator + (const info &l, const info &r)
{
	return {max(l.maxv, r.maxv)};
}

info operator + (const info &v, const tag &t)
{
	return {v.maxv + t.add};
}

tag operator + (const tag &t1, const tag &t2)
{
	return {t1.add + t2.add};
}

struct node
{
	tag t;
	info val;
}tr[N * 4];

void pushup(int u)
{
	tr[u].val = tr[u << 1].val + tr[u << 1 | 1].val;
}

void settag(int u, tag t)
{
	tr[u].val = tr[u].val + t;
	tr[u].t = tr[u].t + t;
}

void pushdown(int u)
{
	if(tr[u].t.add != 0)
	{
		settag(u << 1, tr[u].t);
		settag(u << 1 | 1, tr[u].t);
		tr[u].t.add = 0;
	}
}

void build(int u, int l, int r)
{
	if(l == r)	tr[u].val = {a[l]};
	else
	{
		ll mid = l + r >> 1;
		build(u << 1, l, mid);
		build(u << 1 | 1, mid + 1, r);
		pushup(u);
	}
}

void modify(int u, int l, int r, int ql, int qr, tag t)
{
	if(l == ql && r == qr)
	{
		settag(u, t);
		return ;
	}
	ll mid = l + r >> 1;
	pushdown(u);
	if(qr <= mid)	modify(u << 1, l, mid, ql, qr, t);
	else if(ql > mid)	modify(u << 1 | 1, mid + 1, r, ql, qr, t);
	else
	{
		modify(u << 1, l, mid, ql, mid, t);
		modify(u << 1 | 1, mid + 1, r, mid + 1, qr, t);
	}
	pushup(u);
}

info query(int u, int l, int r, int ql, int qr)
{
	if(l == ql && r == qr)
		return tr[u].val;
	ll mid = l + r >> 1;
	pushdown(u);
	if(qr <= mid)	query(u << 1, l, mid, ql, qr);
	else if(ql > mid)	query(u << 1 | 1, mid + 1, r, ql, qr);
	else
	{
		return query(u << 1, l, mid, ql, mid) + query(u << 1 | 1, mid + 1, r, mid + 1, qr);
	}
}

int main()
{
	scanf("%d%d", &n, &q);
	for(int i = 1; i <= n; i ++)	scanf("%d", &a[i]);
	build(1, 1, n);
	while(q --)
	{
		int op;
		scanf("%d", &op);
		if(op == 1)
		{
			int l, r, d;
			scanf("%d%d%d", &l, &r, &d);
			modify(1, 1, n, l, r, (tag){d});
		}
		else
		{
			int l, r;
			scanf("%d%d", &l, &r);
			auto a = query(1, 1, n, l, r);
			printf("%lld\n", a.maxv);
		}
	}
	return 0;
}

区间修改，区间查询线段树（复杂标记）

#include<bits/stdc++.h>
using namespace std;
const int N = 2e5 + 10, mod = 1e9 + 7;
typedef long long ll;
ll n, q;
ll a[N];

struct tag
{
	ll mul, add;
};

tag operator + (const tag &t1, const tag &t2)
{
	return {t1.mul * t2.mul % mod, (t1.add * t2.mul + t2.add) % mod};
}

struct node
{
	tag t;
	ll val;
	int siz;
}tr[N * 4];

void pushup(int u)
{
	tr[u].val = (tr[u << 1].val + tr[u << 1 | 1].val) % mod;
}

void settag(int u, tag t)
{
	tr[u].val = (tr[u].val * t.mul + tr[u].siz * t.add) % mod;
	tr[u].t = tr[u].t + t;
}

void pushdown(int u)
{
	if(tr[u].t.mul != 1 || tr[u].t.add != 0)
	{
		settag(u << 1, tr[u].t);
		settag(u << 1 | 1, tr[u].t);
		tr[u].t.add = 0;
		tr[u].t.mul = 1;
	}
}

void build(int u, int l, int r)
{
	tr[u].t = {1, 0};
	tr[u].siz = r - l + 1;
	if(l == r)	tr[u].val = {a[l]};
	else
	{
		ll mid = l + r >> 1;
		build(u << 1, l, mid);
		build(u << 1 | 1, mid + 1, r);
		pushup(u);
	}
}

void modify(int u, int l, int r, int ql, int qr, tag t)
{
	if(l == ql && r == qr)
	{
		settag(u, t);
		return ;
	}
	ll mid = l + r >> 1;
	pushdown(u);
	if(qr <= mid)	modify(u << 1, l, mid, ql, qr, t);
	else if(ql > mid)	modify(u << 1 | 1, mid + 1, r, ql, qr, t);
	else
	{
		modify(u << 1, l, mid, ql, mid, t);
		modify(u << 1 | 1, mid + 1, r, mid + 1, qr, t);
	}
	pushup(u);
}

ll query(int u, int l, int r, int ql, int qr)
{
	if(l == ql && r == qr)
		return tr[u].val;
	ll mid = l + r >> 1;
	pushdown(u);
	if(qr <= mid)	query(u << 1, l, mid, ql, qr);
	else if(ql > mid)	query(u << 1 | 1, mid + 1, r, ql, qr);
	else
	{
		return (query(u << 1, l, mid, ql, mid) + query(u << 1 | 1, mid + 1, r, mid + 1, qr)) % mod;
	}
}

int main()
{
	scanf("%d%d", &n, &q);
	for(int i = 1; i <= n; i ++)	scanf("%d", &a[i]);
	build(1, 1, n);
	while(q --)
	{
		int op;
		scanf("%d", &op);
		if(op <= 3)
		{
			int l, r, d;
			scanf("%d%d%d", &l, &r, &d);
			if(op == 1)
				modify(1, 1, n, l, r, (tag){1, d});
			if(op == 2)
				modify(1, 1, n, l, r, (tag){d, 0});
			if(op == 3)
				modify(1, 1, n, l, r, (tag){0, d});
		}
		else
		{
			int l, r;
			scanf("%d%d", &l, &r);
			ll res = query(1, 1, n, l, r);
			printf("%lld\n", res);
		}
	}
	return 0;
}
