网络流基础

网络流

1. 基本概念

1.1 最大流最小割定理：

​ （1）流fff是最大流

​ （2）流fff的残留网络中不存在增广路

​ （3）存在某个割[S,T],∣f∣=c(S,T)[S,T],|f| = c(S,T)[S,T],∣f∣=c(S,T)

​ 以上条件满足任何一个，其余条件均成立

​ 证明：由（1）推理（2），假设流fff的残留网络中存在增广路f′f'f′，则可以根据f′f'f′与流fff构造更大的流，与fff是最大流矛盾，故（1）可以推出（2）

由（2）推（3），设SSS：在残留网络GfG_fGf​中，从sss出发沿着容量大于0的边走，所有能走到的点。

设T=V−ST = V - ST=V−S，得到一个合法的割。

对于原网络流中任意由SSS到TTT的边(x,y)(x,y)(x,y)，一定满足f(x,y)=c(x,y)f(x,y) = c(x,y)f(x,y)=c(x,y)，否则其残留网络的值不为0，与SSS和TTT的定义矛盾

对于原网络流中任意由TTT到SSS的边(a,b)(a, b)(a,b)，一定满足f(a,b)=0f(a,b) = 0f(a,b)=0，否则其残留网络会产生一条由SSS到TTT的反向边，值为原网络流中TTT到SSS的容量，与SSS和TTT的定义矛盾。

于是有∣f∣=f(S,T)=∑u∈S∑v∈Tf(u,v)−0=∑u∈S∑v∈Tc(u,v)=c(S,T)|f| = f(S,T) = \sum_{u\in S} \sum_{v \in T}f(u,v) - 0 = \sum_{u\in S} \sum_{v \in T}c(u,v) = c(S,T)∣f∣=f(S,T)=∑u∈S​∑v∈T​f(u,v)−0=∑u∈S​∑v∈T​c(u,v)=c(S,T)

由（3）推（1），因为∣f∣≤c(S,T)|f| \leq c(S,T)∣f∣≤c(S,T)，设最大流为f′f'f′，则∣f′∣≤c(S,T)|f'| \leq c(S,T)∣f′∣≤c(S,T)，之后证明∣f∣=∣f′∣|f| = |f'|∣f∣=∣f′∣，根据定义显然∣f′∣≥∣f∣|f'| \geq |f|∣f′∣≥∣f∣，由于存在某个割使得∣f∣=c(S,T)|f|=c(S,T)∣f∣=c(S,T)，而∣f′∣≤c(S,T)|f'| \leq c(S,T)∣f′∣≤c(S,T)，因此∣f∣≥∣f′∣|f| \geq |f'|∣f∣≥∣f′∣，由此可得∣f∣=∣f′∣|f| = |f'|∣f∣=∣f′∣，即流fff是最大流

证毕。

2. EK算法求最大流

维护一个残留网络，每次尝试从SSS找到一条到达TTT的增广路径，将其流量从原网络流中移除，同时维护新的残留网络，将这条路径上的最小残留量（流量）累加到resresres中，直到无法找到增广路径，此时resresres即为最大流

时间复杂度O(nm2)O(nm^2)O(nm2)

模板 
#include<bits/stdc++.h>
using namespace std;
const int N = 1010, M = 2e5 + 10, INF = 1e8;
int h[N], e[M], ne[M], f[M], idx;
int d[N], pre[N];
int res;
bool st[N];
int n, m, S, T;

void add(int a, int b, int c)
{
	e[idx] = b, ne[idx] = h[a], f[idx] = c, h[a] = idx ++;
	e[idx] = a, ne[idx] = h[b], f[idx] = 0, h[b] = idx ++;
}

bool bfs()
{
	queue<int> q;
	memset(st, 0, sizeof st);
	q.push(S);
	d[S] = INF;
	st[S] = true;
	while(q.size())
	{
		int t = q.front();
		q.pop();
		for(int i = h[t]; ~i; i = ne[i])
		{
			int j = e[i];
			if(!st[j] && f[i])
			{
				st[j] = true;
				d[j] = min(d[t], f[i]);
				pre[j] = i;
				if(j == T)	return true;
				q.push(j);
			}
		}
	}
	return false;
}

void EK()
{
	while(bfs())
	{
		res += d[T];
		for(int i = T; i != S; i = e[pre[i] ^ 1])
		{
			f[pre[i]] -= d[T];
			f[pre[i] ^ 1] += d[T];
		}
	}
}

int main()
{
	memset(h, -1, sizeof h);
	cin >> n >> m >> S >> T;
	while(m --)
	{
		int a, b, c;
		cin >> a >> b >> c;
		add(a, b, c);
	}
	EK();
	cout << res << endl;
	return 0;
}

3.Dinic算法求最大流

对EKEKEK算法的优化，主要采用分层图+当前弧优化的方式，在求增广路径时可以一次求出多条增广路从而降低时间复杂度，Dinic算法的时间复杂度为O(n2m)O(n^2m)O(n2m)

当前弧优化：在通过DFSDFSDFS遍历当前点到终点的流量时，如果某条路径已经被遍历过且流量已满，则跳过对这条路径的DFSDFSDFS，直接从第一个未满的路径搜索，避免了重复搜索

模板 
#include<bits/stdc++.h>
using namespace std;
const int N = 10010, M = 2e5 + 10, INF = 1e9;
int h[N], e[M], ne[M], f[M], idx;
int d[N], cur[N];
int res, flow;
int n, m, S, T;

void add(int a, int b, int c)
{
	e[idx] = b, ne[idx] = h[a], f[idx] = c, h[a] = idx ++;
	e[idx] = a, ne[idx] = h[b], f[idx] = 0, h[b] = idx ++;
}

bool bfs()
{
	queue<int> q;
	memset(d, -1, sizeof d);
	q.push(S);
	d[S] = 0;
	cur[S] = h[S];		//当前弧从第一个点开始
	while(q.size())
	{
		int t = q.front();
		q.pop();
		for(int i = h[t]; ~i; i = ne[i])
		{
			int j = e[i];
			if(d[j] == -1 && f[i])
			{
				d[j] = d[t] + 1;
				cur[j] = h[j];	//当前点的当前弧是当前点的第一条边
				if(j == T)	return true;
				q.push(j);
			}
		}
	}
	return false;
}

int dfs(int u, int l)
{
	if(u == T)	return l;
	int flow = 0;
	for(int i = cur[u]; ~i && flow < l; i = ne[i])
	{
		cur[u] = i;		//当前弧优化
		int j = e[i];
		if(d[j] == d[u] + 1 && f[i])
		{
			int t = dfs(j, min(f[i], l - flow));
			if(!t)	d[j] = -1;		//不存在增广路，将这个点删掉
			f[i] -= t, f[i ^ 1] += t, flow += t;
		}
	}
	return flow;
}

void dinic()
{
	while(bfs())
	{
		while(flow = dfs(S, INF))	res += flow;
	}
}

int main()
{
	memset(h, -1, sizeof h);
	cin >> n >> m >> S >> T;
	while(m --)
	{
		int a, b, c;
		cin >> a >> b >> c;
		add(a, b, c);
	}
	dinic();
	cout << res << endl;
	return 0;
}

4. 网络流求解问题的步骤

  1. 考虑可行解的集合和流网络的所有可行流的集合
  2. 证明可行解的集合与流网络的所有可行流的集合一一对应（可行流<->可行解）

网络流求解二分图匹配问题（时间复杂度O(mnm\sqrt{n}mn​，匈牙利算法为O(nm))

匈牙利算法本质上是EK算法，二分图匹配问题即在网络流中求最大流

模板（带具体方案） 
#include<bits/stdc++.h>
using namespace std;
const int N = 100, M = 5210, INF = 1e8;
int n, m, S, T;
int h[N], e[M], ne[M], f[M], idx;
int d[N], cur[N];
int res, flow;

void add(int a, int b, int c)
{
    e[idx] = b, ne[idx] = h[a], f[idx] = c; h[a] = idx ++;
    e[idx] = a, ne[idx] = h[b], f[idx] = 0, h[b] = idx ++;
}

bool bfs()
{
    queue<int> q;
    q.push(S);
    memset(d, -1, sizeof d);
    d[S] = 0, cur[S] = h[S];
    while(q.size())
    {
        int t = q.front();
        q.pop();
        for(int i = h[t]; ~ i; i = ne[i])
        {
            int j = e[i];
            if(d[j] == -1 && f[i])
            {
                d[j] = d[t] + 1;
                cur[j] = h[j];
                if(j == T)  return true;
                q.push(j);
            }
        }
    }
    return false;
}

int dfs(int u, int l)
{
    if(u == T)  return l;
    int flow = 0;
    for(int i = cur[u]; ~ i && flow < l; i = ne[i])
    {
        cur[u] = i;
        int j = e[i];
        if(d[j] == d[u] + 1 && f[i])
        {
            int t = dfs(j, min(f[i], l - flow));
            if(! t) d[j] = -1;
            f[i] -= t, f[i ^ 1] += t, flow += t;
        }
    }
    return flow;
}

int dinic()
{
    int r = 0, flow;
    while(bfs())
    {
        while(flow = dfs(S, INF))   r += flow;
    }
    return r;
}

int main()
{
    cin >> m >> n;
    S = 0, T = n + 1;
    memset(h, -1, sizeof h);
    for(int i = 1; i <= m; i ++)    add(S, i, 1);
    for(int i = m + 1; i <= n; i ++)    add(i, T, 1);
    int a, b;
    while(cin >> a >> b, a != -1)   add(a, b, 1);
    cout << dinic() << endl;
    for(int i = 0; i < idx; i += 2)
    {
        if(e[i] > m && e[i] <= n && !f[i])
            cout << e[i ^ 1] << " " << e[i] << endl;
    }
    return 0;
}

5. 上下界可行流

具有上界和下界限制的可行流，同样满足流量守恒，但是容量具有限制

![https://s3.bmp.ovh/imgs/2023/01/23/6e82ededdda13827.jpg)

模板题：

给定一个包含 nnn 个点 mmm 条边的有向图，每条边都有一个流量下界和流量上界，求一种可行方案使得在所有点满足流量平衡条件的前提下，所有边满足流量限制。

模板 
#include<bits/stdc++.h>
using namespace std;
const int N = 600, M = 1e6, INF = 1e8;
int h[N], e[M], ne[M], f[M], idx, l[M];
int cur[N], d[N], A[N];
int m, n, S, T;

void add(int a, int b, int c, int d)
{
    e[idx] = b, ne[idx] = h[a], f[idx] = d - c, l[idx] = c, h[a] = idx ++;
    e[idx] = a, ne[idx] = h[b]; f[idx] = 0, h[b] = idx ++;
}

bool bfs()
{
    memset(d, -1, sizeof d);
    queue<int> q;
    q.push(S);
    d[S] = 0, cur[S] = h[S];
    while(q.size())
    {
        int t = q.front();
        q.pop();
        for(int i = h[t]; ~i; i = ne[i])
        {
            int j = e[i];
            if(d[j] == -1 && f[i])
            {
                d[j] = d[t] + 1;
                cur[j] = h[j];
                if(j == T)  return true;
                q.push(j);
            }
        }
    }
    return false;
}

int dfs(int u, int l)
{
    if(u == T)  return l;
    int flow = 0;
    for(int i = cur[u]; ~i && flow < l; i = ne[i])
    {
        cur[u] = i;
        int j = e[i];
        if(d[j] == d[u] + 1 && f[i])
        {
            int t = dfs(j, min(f[i], l - flow));
            if(!t)  d[j] = -1;
            f[i] -= t, f[i ^ 1] += t, flow += t;
        }
    }
    return flow;
}

int dinic()
{
    int r = 0, flow;
    while(bfs())
    {
        while(flow = dfs(S, INF))   r += flow;
    }
    return r;
}

int main()
{
    memset(h, -1, sizeof h);
    cin >> n >> m;
    int sum = 0;
    S = 0, T = n + 1;
    for(int i = 0; i < m; i ++)
    {
        int a, b, c, d;
        cin >> a >> b >> c >> d;
        add(a, b, c, d);
        A[a] -= c, A[b] += c;
    }

    for(int i = 1; i <= n; i ++)
    {
        if(A[i] > 0)    add(S, i, 0, A[i]), sum += A[i];
        else if(A[i] < 0)   add(i, T, 0, -A[i]);
    }

    if(dinic() != sum)  cout << "NO" << endl;
    else
    {
        cout << "YES" << endl;
        for(int i = 0; i < m * 2; i += 2)
            cout << f[i ^ 1] + l[i] << endl;
    }
    return 0;
}
