1721D -Maximum AND

题目大意:给出两个长度为nnn的数组a,ba,ba,b，规定f(a,b)f(a,b)f(a,b)表示以下操作：

  1. 创建一个长度为nnn的数组，满足ci=ai⊕bic_i=a_i \oplus b_ici​=ai​⊕bi​
  2. 则 f(a,b)=c1&c2&...&cnf(a,b) = c_1 \& c_2 \&...\&c_nf(a,b)=c1​&c2​&...&cn​

你可以将bbb数组进行任意的排列，求最大的f(a,b)f(a,b)f(a,b)

数据范围：1≤n≤1051 \leq n \leq 10^51≤n≤105

题解：二进制构造题，首先考虑二进制拆位，由于f(a,b)f(a,b)f(a,b)由按位与操作得到，因此某一位为111必然有：

∀ci\forall c_i∀ci​，该位的二进制为111

又由于ci=ai⊕bic_i=a_i \oplus b_ici​=ai​⊕bi​，则ai与bia_i 与 b_iai​与bi​必然在该位上满足：000的个数与111的个数相同

由此考虑贪心：从高位向低位枚举，如果满足该位上000与111的个数相同，就选取这一位，但是这种贪心方式是错误的，因为：

2
3 0
2 1

如果按上述方式选取，会发现bbb中一定有一个数被选择了两次，这显然是不合法的

于是进一步考虑一个数能被选取的条件，不妨设答案为resresres

如果存在bbb的排序，∀ai\forall a_i∀ai​有(ai&res)⊕(bi&res)=res(a_i \& res) \oplus (b_i \& res) = res(ai​&res)⊕(bi​&res)=res（即前iii位中aaa中1的个数与bbb中0的个数相同），则resresres即为答案。

我们考虑每一位，将其异或resresres,并将两个得到的数组排序，如果相同，则该位为111，否则为000

// Problem: D. Maximum AND
// Contest: Codeforces - Educational Codeforces Round 134 (Rated for Div. 2)
// URL: https://codeforces.com/problemset/problem/1721/D
// Memory Limit: 256 MB
// Time Limit: 3000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#define int long long
using namespace std;
const int N = 2e5 + 10;
int n;
int a[N];
int b[N];

bool check(int x)
{
	vector<int> c, d;
	for(int i = 1; i <= n; i ++)
	{
		c.push_back(a[i] & x);
		d.push_back((b[i] & x) ^ x);
	}
	sort(c.begin(), c.end());
	sort(d.begin(), d.end());
	return c == d;
}

void solve()
{
	cin >> n;
	int res = 0;
	for(int i = 1; i <= n; i ++)	cin >> a[i];
	for(int i = 1; i <= n; i ++)	cin >> b[i];
	for(int i = 30; i >= 0; i --)
	{
		if(check(res | (1 << i)))	res |= (1 << i);
	}
	cout << res << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	int T;
	cin >> T;
	while(T --)
	{
		solve();
	}

}

1718A1/A2 - Burenka and Traditions (easy version)+(hard version)

题目大意：有一个长度为nnn的数组aaa，每一次操作可以选择一段区间[l,r][l,r][l,r]和一个非负整数xxx，花费[(r−l+1)/2][(r - l + 1) / 2][(r−l+1)/2]秒使区间内的数都异或xxx，问最少几秒才能把数组中所有元素变成000

数据范围：1≤n≤50001 \leq n \leq 50001≤n≤5000，0≤ai≤50000 \leq a_i \leq 50000≤ai​≤5000

题解：我们可以对操作做等价变形，我们可以发现操作其实可以分为两种：

  1. 异或一个数，代价为111
  2. 异或连续两个数，代价为111

显然，我们至多对每个数进行一次第一种操作，答案至多为nnn，考虑第二种操作对答案的影响，可以发现，如果一段连续的数异或为000，那么就可以连续用第二种操作，每用一次第二种操作，答案就会减少111，因此我们可以维护所有异或和为000的块，这启发我们考虑前缀和。

对原数组求前缀异或和sumsumsum，如果sum(1,l)=sum(1,r)sum(1,l)=sum(1,r)sum(1,l)=sum(1,r)，说明此时可以使用一次第二种操作，答案减少111，我们可以用mapmapmap来维护前缀异或和的值是否出现过，注意特判000的情况即可。

// Problem: A1. Burenka and Traditions (easy version)
// Contest: Codeforces - Codeforces Round #814 (Div. 1)

		int x;
		cin >> x;
		seg ^= x;

		{
			res --;
			m.clear();
			m[0] = 1;

		}
		else	m[seg] = 1;

1718B - Fibonacci Strings

题目大意：给定一个数列ck{c_k}ck​

你可以执行若干次操作，对于第iii次操作：

  1. 选定一个[1,n][1,n][1,n]范围内的整数did_idi​，当i≥2i \geq 2i≥2时，必须保证di≠di−1d_i \neq d_{i - 1}di​​=di−1​。
  2. 将cdic_{d_i}cdi​​减去fif_ifi​，其中fif_ifi​是斐波那契数列中第iii项

问：c1c_1c1​至ckc_kck​能否全为0？

数据范围：1≤k≤1061 \leq k \leq 10^61≤k≤106

题解：首先可以观察到，只有当∑i=1kci\sum_{i = 1}^k c_i∑i=1k​ci​等于斐波那契数列某项的前缀和时，才有可能全为0

之后考虑如何选择最优解

  1. 优先考虑插入最小的数到最小的斐波那契数
  2. 优先考虑插入最大的数到最大的斐波那契数

显然方案2为最优解，注意特判每次不能选两个相同的下标插入即可

// Problem: B. Fibonacci Strings
// Contest: Codeforces - Codeforces Round #814 (Div. 1)
// URL: https://codeforces.com/problemset/problem/1718/B
// Memory Limit: 256 MB
// Time Limit: 2000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#include<queue>
#define int long long
using namespace std;
const int N = 2e5 + 10;
typedef pair<int, int> PII;
int a[N];
int feb[N];
int sum[N];
int idx = 0;

void init()
{
	feb[1] = 1;
	feb[2] = 1;
	sum[1] = 1;
	sum[2] = 2;
	for(int i = 3; i <= 50; i ++)
	{
		feb[i] = feb[i - 1] + feb[i - 2];
		sum[i] = sum[i - 1] + feb[i];
	}
}

bool check(int s)
{
	for(int i = 1; i <= 50; i ++)
	{
		if(sum[i] == s)
		{
			idx = i;
			return true;
		}
	}
	return false;
}

void solve()
{
	priority_queue<PII, vector<PII> > heap;
	idx = 0;
	int n;
	cin >> n;
	int s = 0;
	for(int i = 1; i <= n; i ++)
	{
		cin >> a[i];
		s += a[i];
		heap.push({a[i], i});
	}
	bool flag = 1;
	if(!check(s))
	{
		cout << "NO" << endl;
		return ;
	}
	int back = -1;
	for(int i = idx; i >= 1; i --)
	{
		auto t = heap.top();
		heap.pop();
		if(t.second == back && heap.size() && heap.top().first != t.first)	flag = 0;
		else if(t.second == back && heap.size() && heap.top().first == t.first)
		{
			auto tt = heap.top();
			heap.pop();
			heap.push(t);
			t = tt;
		}
		else if(t.second == back && heap.empty())	flag = 0;
		
		if(t.first >= feb[i])
		{
			t.first -= feb[i];
			heap.push(t);
			back = t.second;
		}
		else
		{
			flag = 0;
			break;
		}
	}
	if(flag)	cout << "YES" << endl;
	else	cout << "NO" << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	int T;
	cin >> T;
	init();
	while(T --)
	{
		solve();
	}
	return 0;
}

// Problem: D. Chip Move
// Contest: Codeforces - Educational Codeforces Round 133 (Rated for Div. 2)
// URL: https://codeforces.com/problemset/problem/1716/D
// Memory Limit: 256 MB
// Time Limit: 2000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#define int long long
using namespace std;
const int N = 2e5 + 10, mod = 998244353;
int dp[2][N], res[N];

void solve()
{
	int n, k;
	cin >> n >> k;
	int sum = k, id = 1;
	dp[0][0] = 1;
	for(int i = k; sum <= n; i ++)
	{
		id ^= 1;
		for(int j = i; j <= n; j ++)
			dp[id][j] = (dp[id][j] + dp[id ^ 1][j - i] + dp[id][j - i]) % mod;
		dp[id][0] = 0, sum += i;
		for(int j = 1; j <= n; j ++)
		{
			res[j] = (res[j] + dp[id][j]) % mod;
			dp[id ^ 1][j] = 0;
		}
	}
	for(int i = 1; i <= n; i ++)	cout << res[i] << " ";
	cout << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	solve();
	return 0;
}

1716C - Robot in a Hallway

题目大意：

题解：

// Problem: C. Robot in a Hallway
// Contest: Codeforces - Educational Codeforces Round 133 (Rated for Div. 2)
// URL: https://codeforces.com/problemset/problem/1716/C
// Memory Limit: 256 MB
// Time Limit: 2000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#define int long long
using namespace std;
const int N = 2e5 + 10;
int a[2][N], f[2][N];
int n;

void solve()
{
	cin >> n;
	for(int i = 0; i <= 1; i ++)
		for(int j = 1; j <= n; j ++)
			cin >> a[i][j];
	a[0][1] = -1;
	f[0][n] = a[0][n], f[1][n] = a[1][n];
	for(int i = 0; i <= 1; i ++)
		for(int j = n - 1; j >= 1; j --)
			f[i][j] = max(f[i][j + 1] - 1, a[i][j]);
	f[0][n] = max(a[1][n] - 1, a[0][n]);
	f[1][n] = max(a[0][n] - 1, a[1][n]);
	for(int i = 0; i <= 1; i ++)
		for(int j = n - 1; j >= 1; j --)
			f[i][j] = max(f[i][j + 1] - 1, max(a[i][j], a[i ^ 1][j] - 2 * (n - j) - 1));
	int res = f[0][1] + 2 * n, rec = a[1][1] + 1;;
	for(int i = 2, p = 1; i <= n; i ++, p ^= 1)
	{
		res = min(res, rec + (f[p][i] - rec > 0 ? f[p][i] - rec : 0ll) + 2 * (n - i + 1));
		rec = max(rec + 1, a[p][i] + 1);
		rec = max(rec + 1, a[p ^ 1][i] + 1);
	}
	res = min(res, rec);
	cout << res << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	int T;
	cin >> T;
	while(T --)
	{
		solve();
	}
	return 0;
}

// Problem: C. Zero Path
// Contest: Codeforces - Codeforces Round #801 (Div. 2) and EPIC Institute of Technology Round
// URL: https://codeforces.com/problemset/problem/1695/C

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#define int long long
using namespace std;
const int N = 1010, INF = 0x3f3f3f3f;
int a[N][N];
int pd[N][N];
int dp[N][N];

void solve()
{
	int n, m;
	cin >> n >> m;
	for(int i = 1; i <= n; i ++)
		for(int j = 1; j <= m; j ++)
			cin >> a[i][j];
	if((n + m - 1) & 1)
	{
		cout << "NO" << endl;
		return ;
	}
	int h = (n + m - 1) / 2;
	dp[1][1] = (a[1][1] == 1);
	for(int i = 1; i <= n; i ++)
		for(int j = 1; j <= m; j ++)
			dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]) + (a[i][j] == 1);
	for(int i = 2; i <= n; i ++)	pd[i][0] = INF;
	for(int i = 2; i <= m; i ++)	pd[0][i] = INF;
	for(int i = 1; i <= n; i ++)
		for(int j = 1; j <= m; j ++)
			pd[i][j] = min(pd[i - 1][j], pd[i][j - 1]) + (a[i][j] == 1);
	if(h >= pd[n][m] && h <= dp[n][m])	cout << "YES" << endl;

// Problem: C. Bakry and Partitioning
// Contest: Codeforces - Codeforces Round #746 (Div. 2)
// URL: https://codeforces.com/problemset/problem/1592/C
// Memory Limit: 256 MB
// Time Limit: 1000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#define int long long
using namespace std;
const int N = 2e5 + 10, M = N * 2;
int h[N], e[M], ne[M], w[N], idx;
int n, k;
int dp[N];
bool v[N];

void add(int a, int b)
{
	e[idx] = b;
	ne[idx] = h[a];
	h[a] = idx ++;
}

void dfs(int u, int val)
{
	dp[u] = w[u];
	for(int i = h[u]; ~i; i = ne[i])
	{
		int j = e[i];
		if(v[j])	continue;
		v[j] = true;
		dfs(j, val);
		if(dp[j] != val)	dp[u] ^= dp[j];
	}
}

void solve()
{
	cin >> n >> k;
	memset(v, 0, sizeof v);
	idx = 0;
	int sum = 0;
	for(int i = 1; i <= n; i ++)
	{
		h[i] = -1;
		dp[i] = 0;
		cin >> w[i];
		sum ^= w[i];
	}	
	for(int i = 1; i <= n - 1; i ++)
	{
		int a, b;
		cin >> a >> b;
		add(a, b);
		add(b, a);
	}
	if(sum == 0)
	{
		cout << "YES" << endl;
		return;
	}
	else if(k == 2)
	{
		cout << "NO" << endl;
		return;
	}
	dfs(1, sum);
	int cnt = 0;
	for(int i = 1; i <= n; i ++)
		cnt += dp[i] == sum;
	if(cnt >= 2)	cout << "YES" << endl;
	else	cout << "NO" << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	int T;
	cin >> T;
	while(T --)
	{
		solve();
	}
	return 0;

1454 - E Number of Simple Paths

题目大意：给定一颗基环树，求基环树中长度大于等于111的不同简单路径的数量

题解:

// Problem: E. Number of Simple Paths
// Contest: Codeforces - Codeforces Round #686 (Div. 3)
// URL: https://codeforces.com/problemset/problem/1454/E
// Memory Limit: 256 MB
// Time Limit: 2000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#include<queue>
#define int long long
using namespace std;
const int N = 2e5 + 10, M = N * 2;
int h[N], e[M], ne[M], idx;
int siz[N];
int n;
bool v[N];
int d[N];

void add(int a, int b)
{
	e[idx] = b;
	ne[idx] = h[a];
	h[a] = idx ++;
}

void topsort()
{
	queue<int> q;
	for(int i = 1; i <= n; i ++)
		if(d[i] == 1)	q.push(i);
	while(q.size())
	{
		int t = q.front();
		q.pop();
		for(int i = h[t]; ~i; i = ne[i])
		{
			int j = e[i];
			if(-- d[j] == 1)	q.push(j);
		}
	}
}

void dfs(int u, int fa)
{
	siz[u] = 1;
	for(int i = h[u]; ~i; i = ne[i])
	{
		int j = e[i];
		if(j != fa)
		{ 
			dfs(j, u);
			siz[u] += siz[j];
		}
	}
}

void solve()
{
	cin >> n;
	int res = 0, t = 0;;
	for(int i = 1; i <= n; i ++)
	{
		h[i] = -1;
		v[i] = 0;
		siz[i] = 0;
		idx = 0;
		d[i] = 0;
	}
	for(int i = 1; i <= n; i ++)
	{
		int a, b;
		cin >> a >> b;
		add(a, b);
		add(b, a);
		d[a] ++;
		d[b] ++;
	}
	topsort();
	for(int i = 1; i <= n; i ++)
		if(d[i] > 1)	v[i] = true;
	for(int i = 1; i <= n; i ++)
	{
		if(v[i])
		{
			t = 0;
			for(int j = h[i]; ~j; j = ne[j])
			{
				int p = e[j];
				if(!v[p])
				{
					dfs(p, i);
					t += siz[p];
				}
			}
			res += t;
			res += t * (t - 1) / 2;
		}
	}
	cout << n * (n - 1) - res << endl;
}

signed main()

// Problem: G. Unusual Minesweeper
// Contest: Codeforces - Codeforces Round #762 (Div. 3)
// URL: https://codeforces.com/problemset/problem/1619/G
// Memory Limit: 256 MB
// Time Limit: 2000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#define int long long
using namespace std;
const int N = 2e5 + 10, INF = 1e18;
typedef pair<int, int> PII;
int n, k;
pair<pair<int, int>, int> poi[N];
int p[N], t[N];
int res[N];
int x, y, id, l, i, ii, nn;

int find(int x)
{
	if(p[x] != x)	p[x] = find(p[x]);
	return p[x];
} 


void solve()
{
	cin >> n >> k;
	for(i = 1; i <= n; i ++)
	{
		cin >> x >> y >> t[i];
		poi[i] = {{x, y}, i};
		p[i] = i;
	}
	for(ii = 0; ii < 2; ii ++)
	{
		sort(poi + 1, poi + 1 + n);
		for(i = 1; i <= n; i ++)
		{
			y = poi[i].first.first;
			x = poi[i].first.second;
			id = poi[i].second;
			if(i > 1 && y == poi[i - 1].first.first && x - poi[i - 1].first.second <= k)
			{
				l = poi[i - 1].second;
				t[find(id)] = min(t[find(id)], t[find(l)]);
				p[find(l)] = find(id);
			}
		}
		for(int i = 1; i <= n; i ++)	swap(poi[i].first.first, poi[i].first.second);
	}
	nn = 0;
	for(i = 1; i <= n; i ++)
	{
		if(find(i) == i)
		{
			nn ++;
			res[nn] = t[i];
		}
	} 
	sort(res + 1, res + nn + 1, greater<int> ());
	res[nn + 1] = -INF;
	for(i = 1; i <= nn; i ++)
		if(res[i + 1] <= i - 1)	break;
	cout << i - 1 << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	int T;
	cin >> T;
	while(T --)

1722F - L-shapes

题目大意：

题解：若一个点是L形的一部分，先用BFS四连通找此连通块是否是L形，再找此点的八连通看是否只有两个L形的一部分，若不是，则不合法。

// Problem: F. L-shapes
// Contest: Codeforces - Codeforces Round #817 (Div. 4)
// URL: https://codeforces.com/problemset/problem/1722/F
// Memory Limit: 256 MB
// Time Limit: 1000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
#include<queue>
#define int long long
using namespace std;
const int N = 110;
char g[N][N];
int n, m;
bool v[N][N];
int dx4[4] = {1, -1, 0, 0}, dy4[4] = {0, 0, 1, -1};
int dx8[8] = {1, 1, 1, 0, 0, -1, -1, -1}, dy8[8] = {-1, 0, 1, 1, -1, -1, 0, 1};

int bfs_4(int sx, int sy)	//判断L型
{
	if(v[sx][sy])	return 3;
	int cnt = 0;
	v[sx][sy] = 1;
	queue<pair<int, int>> q;
	q.push({sx, sy});
	while(q.size())
	{
		int x = q.front().first, y = q.front().second;
		q.pop();
		cnt ++;
		for(int i = 0; i < 4; i ++)
		{
			int zx = x + dx4[i], zy = y + dy4[i];
			if(zx && zy && zx <= n && zy <= m && !v[zx][zy] && g[zx][zy] == '*' && abs(zx - sx) <= 1 && abs(zy - sy) <= 1)
			{
				v[zx][zy] = 1;
				q.push({zx, zy});
			}
		}
	}
	return cnt;
}

int bfs_8(int sx, int sy)
{
	int cnt = 0;
	for(int i = 0; i < 8; i ++)
	{
		int zx = sx + dx8[i], zy = sy + dy8[i];
		if(zx && zy && zx <= n && zy <= m && g[zx][zy] == '*')	cnt ++;
	}
	return cnt;
}

void solve()
{
	memset(v, 0, sizeof v);
	cin >> n >> m;
	for(int i = 1; i <= n; i ++)
		for(int j = 1; j <= m; j ++)
			cin >> g[i][j];
	for(int i = 1; i <= n; i ++)
		for(int j = 1; j <= m; j ++)
			if(g[i][j] == '*')
			{
				int cnt4 = bfs_4(i, j);
				int cnt8 = bfs_8(i, j);
				if(cnt4 != 3 || cnt8 != 2)
				{
					cout << "NO" << endl;
					return ;
				}
			}
	cout << "YES" << endl;
}

signed main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr), cout.tie(nullptr);
	int T;
	cin >> T;
	while(T --)
	{
		solve();
	}
	return 0;
}

期望DP

将第二种奖品看成容积为−1-1−1，之后考虑DP

dp[i][j][k]dp[i][j][k]dp[i][j][k]表示从前iii次比赛中获胜kkk次，容积为kkk时的概率

状态转移方程：

如果第i+1i+1i+1场比赛输掉，则有dp[i+1][j][k]+=dp[i][j][k]∗(1−p[i+1])dp[i+1][j][k] += dp[i][j][k] * (1 - p[i + 1])dp[i+1][j][k]+=dp[i][j][k]∗(1−p[i+1])

如果第i+1i + 1i+1场比赛获胜，则有dp[i+1][j+1][k+a[i+1]+=dp[i][j][k]∗p[i+1]dp[i+1][j+1][k+a[i+1] += dp[i][j][k] * p[i + 1]dp[i+1][j+1][k+a[i+1]+=dp[i][j][k]∗p[i+1]

// Problem: B. Wizards and Huge Prize

#include<vector>
#include<cmath>

{

题解：

01背包

第一问答案显然，考虑第二问

设dp[i][j][k]dp[i][j][k]dp[i][j][k]为前iii个瓶子里取kkk个瓶子，总容积为jjj的最大不倒入杯中的体积

状态转移方程：dp[i][j][k]=max(dp[i−1][j][k],dp[i−1][j−bi][k]+ai)dp[i][j][k] = max(dp[i - 1][j][k], dp[i - 1][j - b_i][k] + a_i)dp[i][j][k]=max(dp[i−1][j][k],dp[i−1][j−bi​][k]+ai​)

// Problem: J. Bottles
// Contest: Codeforces - 2016-2017 ACM-ICPC, NEERC, Southern Subregional Contest (Online Mirror, ACM-ICPC Rules, Teams Preferred)
// URL: https://codeforces.com/problemset/problem/730/J
// Memory Limit: 512 MB
// Time Limit: 2000 ms
// Author:Ray

#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<string>
using namespace std;
typedef pair<int, int> PII;
const int N = 110;
int res, sum1, sum2;
int n;
int dp[N * N][N];
PII a[N];
int k;

bool cmp(PII a, PII b)
{
	return a.second > b.second;
}

void work()
{
	memset(dp, 128, sizeof dp);
	dp[0][0] = 0;
	for(int i = 1; i <= n; i ++)
		for(int j = sum2; j >= a[i].second; j --)
			for(int p = 1; p <= k; p ++)
				dp[j][p] = max(dp[j][p], dp[j - a[i].second][p - 1] + a[i].first);
	for(int i = sum1; i <= sum2; i ++)
		res = max(res, dp[i][k]);
	cout << sum1 - res << endl;
}

void solve()
{
	cin >> n;
	for(int i = 1; i <= n; i ++)
	{
		cin >> a[i].first;
		sum1 += a[i].first;
	}
	for(int i = 1; i <= n; i ++)
	{
		cin >> a[i].second;
	}
	sort(a + 1, a + n + 1, cmp);
	while(sum2 < sum1)	sum2 += a[++ k].second;
	cout << k << " ";
	work();
}

signed main()
{

}
